# 3291. 形成目标字符串需要的最少字符串数 I

> **作者:** 弘树
> **日期:** 2024-09-15
> **所用时间:** min

## 解题思路
### 1.Trie树



- 时间复杂度：$O()$
- 空间复杂度：$O()$

```C++
class Solution {
public:
    int son[5010][26], idx;

    void add(string str)
    {
        int p = 0;
        for (int i = 0; str[i]; i ++)
        {
            int u = str[i] - 'a';
            if (!son[p][u]) son[p][u] = ++ idx;
            p = son[p][u];
        }
    }

    int query(int i, string str)
    {
        int p = 0;
        while (str[i])
        {
            int u = str[i] - 'a';
            if (!son[p][u]) break;
            p = son[p][u];
            i ++;
        }
        return i;
    }

    int minValidStrings(vector<string>& words, string target) {
        for (auto str : words) add(str);
        int ans = 0, i = 0;
        while (i < target.size())
        {
            int res = query(i, target);
            if (i == res) return -1;
            else i = res, ans ++;
        }
        return ans;
    }
};
```
