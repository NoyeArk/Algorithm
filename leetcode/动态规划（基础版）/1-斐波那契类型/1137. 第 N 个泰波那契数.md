## 解题思路
### 1.动态规划 + 滚动数组

和[朴素斐波那契数列](509.%20斐波那契数.md)类似，多了一个状态，同时对空间进行优化。

- 时间复杂度：O(n)
- 空间复杂度：O(1)

```C++
class Solution {
public:
    int tribonacci(int n) {
        if (n == 0) return 0;
        if (n == 1 || n == 2) return 1;
        int f0 = 0, f1 = 1, f2 = 1, f3;
        for (int i = 3; i <= n; i ++)
            f3 = f0 + f1 + f2, f0 = f1, f1 = f2, f2 = f3;
        return f3;
    }
};
```

### 2.矩阵快速幂

```C++
class Solution {
public:
    typedef long long LL;

    void mult(LL a[][3], LL b[][3])
    {
        LL c[3][3] = { 0 };
        for (int i = 0; i < 3; i ++)
            for (int j = 0; j < 3; j ++)
                for (int k = 0; k < 3; k ++)
                    c[i][j] += a[i][k] * b[k][j];
        memcpy(a, c, sizeof c);
    }

    int tribonacci(int n) {
        LL f[][3] = {{0, 0, 1}, {1, 0, 1}, {0, 1, 1}};
        LL a[][3] = {{0, 1, 1}, {0, 0, 0}, {0, 0, 0}};

        while (n)
        {
            if (n & 1) mult(a, f);
            n >>= 1;
            mult(f, f);
        }

        return a[0][0];
    }
};
```
