## 解题思路
### 1.动态规划

- 时间复杂度：$O(nm)$
- 空间复杂度：$O(nm)$

```C++
class Solution {
public:
    int f[210][210];

    int minPathSum(vector<vector<int>>& grid) {
        int n = grid.size(), m = grid[0].size();
        memset(f, 0x3f, sizeof f);
        f[0][1] = 0;
        for (int i = 1; i <= n; i ++)
            for (int j = 1; j <= m; j ++)
                f[i][j] = min(f[i - 1][j], f[i][j - 1]) + grid[i - 1][j - 1];
        return f[n][m];
    }
};
```
